Fix watchdog plots showing negative fitness values by filtering sentinel values and using correct metrics.

Output: Sentinel-filtered plot data, correct metric usage, arms-race plot format in watchdog.

def generate_plot(hour: int) -> Path | None:
    plot_dir = _LOG_DIR / "plots"
    plot_dir.mkdir(parents=True, exist_ok=True)
    try:
        run_args = []
        for run in RUNS:
            run_args.extend(["-r", f"{run['prefix']}@{run['db']}:{run['label']}"])

        # Build paired spec: G_label:D_label for arms-race plot
        g_labels = [r["label"] for r in RUNS if "pop_a" in r["prefix"]]
        d_labels = [r["label"] for r in RUNS if "pop_b" in r["prefix"]]
        paired_specs = []
        for g, d in zip(g_labels, d_labels):
            paired_specs.append(f"{g}:{d}")
        paired_arg = ",".join(paired_specs)

        subprocess.run(
            ["gigaevo"]
            + run_args
            + [
                "plot",
                "arms-race",
                "--metric", "actual_fitness",
                "--paired", paired_arg,
                "-o",
                str(plot_dir),
            ],
            cwd=str(PROJ),
            capture_output=True,
            timeout=120,
            check=True,
        )
        png = plot_dir / "arms_race.png"  # arms-race output filename
        if png.exists():
            stamped = plot_dir / f"comparison_hour_{hour:03d}.png"
            shutil.copy2(png, stamped)
            log(f"Plot generated: {stamped.name}")
            return stamped
    except Exception as e:
        log(f"Plot generation failed: {e}")
    return None
