# Plan 02-03: GitHubPRChannel with plot upload, rolling comment, cache-busting

<objective>
Implement a `GitHubPRChannel` that posts status tables and plot images to GitHub PR comments.
This replaces the fragile `urllib.request` code in watchdog templates and the basic markdown
builder in `tools/experiment/pr_comment.py`. The new implementation:
(1) uses `httpx.AsyncClient` for the GitHub API (consistent with TelegramChannel),
(2) implements rolling comment: creates a new comment on first post, then edits it on subsequent posts (prevents PR comment spam),
(3) uploads plot PNGs to GitHub Release assets and embeds them with cache-busting `?v={timestamp}` query params,
(4) supports a `telegram_down` flag that prepends a warning header to PR comments when Telegram is failing (used by dispatcher in 02-04).
All code lives in `gigaevo/monitoring/github_pr_channel.py`. No changes to `tools/`.
</objective>

<must_haves>
- `GitHubPRChannel(NotificationChannel)` concrete class with `repo: str`, `pr_number: int`, `token: str`, optional `base_url: str` (for test mocking)
- Uses `httpx.AsyncClient` for all GitHub API calls
- Rolling comment: first `send_status` creates a new PR comment (POST), stores `_comment_id`. Subsequent calls edit that comment (PATCH). The comment ID persists in-memory (lost on restart -- acceptable for v1)
- Plot upload: uploads PNG files to GitHub Release assets. Embeds them in the PR comment body as `![caption](raw_url?v={timestamp})` with cache-busting
- `telegram_down: bool` settable property. When True, PR comment body is prefixed with `> :warning: TELEGRAM DOWN -- notifications are not being delivered to Telegram`
- `send_status(update)`: builds markdown body with header, status table, alerts, plot images. Posts or edits PR comment
- `send_alert(alert)`: posts a new PR comment (always new, never edits -- alerts are discrete events)
- `check_health()`: calls GitHub API `GET /repos/{owner}/{repo}` to verify token and repo access. Returns True on 200
- `format_status_table_markdown` used for the table (NOT-06: same data as Telegram)
- Token sourced from constructor parameter (caller reads from `gh auth token` or env). No direct env var reading in this module
- Logging via `loguru` with `logger.bind(component="github_pr")`
- No imports from `tools/` -- this replaces, not wraps, the existing PR comment code
</must_haves>

```python
"""GitHub PR comment notification channel using httpx.

Replaces the urllib.request code in watchdog templates with:
- httpx.AsyncClient for all GitHub API calls
- Rolling comment (POST once, PATCH thereafter)
- Plot upload to GitHub Release assets with cache-busting URLs
- Cross-channel telegram_down header
"""

from __future__ import annotations

from typing import Any

import httpx
from loguru import logger

from gigaevo.monitoring.alerts import Alert
from gigaevo.monitoring.notifications import (
    NotificationChannel,
    StatusUpdate,
    format_alert_message,
    format_status_table_markdown,
)

_log = logger.bind(component="github_pr")

_DEFAULT_BASE_URL = "https://api.github.com"


class GitHubPRChannel(NotificationChannel):
    """GitHub PR comment notification channel.

    Posts status tables and plot images as PR comments. Uses rolling
    comment pattern: first post creates, subsequent posts edit.
    """

    def __init__(
        self,
        repo: str,
        pr_number: int,
        token: str,
        base_url: str = _DEFAULT_BASE_URL,
        transport: httpx.AsyncBaseTransport | None = None,
    ):
        self._repo = repo
        self._pr_number = pr_number
        self._token = token
        self._base_url = base_url.rstrip("/")
        self._transport = transport
        self._client: httpx.AsyncClient | None = None
        self._comment_id: int | None = None
        self._telegram_down = False

    @property
    def telegram_down(self) -> bool:
        return self._telegram_down

    @telegram_down.setter
    def telegram_down(self, value: bool) -> None:
        self._telegram_down = value

    async def _get_client(self) -> httpx.AsyncClient:
        if self._client is None or self._client.is_closed:
            kwargs: dict[str, Any] = {
                "timeout": httpx.Timeout(30.0),
                "headers": {
                    "Authorization": f"token {self._token}",
                    "Accept": "application/vnd.github.v3+json",
                },
            }
            if self._transport is not None:
                kwargs["transport"] = self._transport
            self._client = httpx.AsyncClient(**kwargs)
        return self._client

    async def close(self) -> None:
        if self._client is not None and not self._client.is_closed:
            await self._client.aclose()
            self._client = None

    async def check_health(self) -> bool:
        """Verify the token and repo are valid by calling GET /repos/{owner}/{repo}."""
        try:
            client = await self._get_client()
            resp = await client.get(f"{self._base_url}/repos/{self._repo}")
            return resp.status_code == 200
        except Exception as exc:
            _log.warning(f"GitHub health check failed: {exc}")
            return False

    # send_status and send_alert implemented in Task 4
    async def send_status(self, update: StatusUpdate) -> bool:
        raise NotImplementedError

    async def send_alert(self, alert: Alert) -> bool:
        raise NotImplementedError
```

```python
    def _build_status_body(self, update: StatusUpdate) -> str:
        """Build the markdown body for a status update PR comment."""
        parts: list[str] = []

        # Telegram down header
        if self._telegram_down:
            parts.append("> :warning: TELEGRAM DOWN -- notifications are not being delivered to Telegram")
            parts.append("")

        # Header
        parts.append(f"### {update.experiment_name}")
        if update.max_generations is not None:
            parts.append(f"Target: {update.max_generations} generations")
        ts = update.timestamp.strftime("%Y-%m-%d %H:%M UTC")
        parts.append(f"_Updated: {ts}_")
        parts.append("")

        # Status table
        table = format_status_table_markdown(update.snapshots)
        parts.append(table)

        # Alerts
        if update.has_alerts:
            parts.append("")
            parts.append("**Alerts:**")
            for alert in update.alerts:
                parts.append(f"- {format_alert_message(alert)}")

        # Plot images (embedded with cache-busting)
        if update.has_plots:
            parts.append("")
            parts.append("**Plots:**")
            for plot in update.plots:
                # Use timestamp for cache-busting
                ts_bust = int(update.timestamp.timestamp())
                if hasattr(plot, '_upload_url') and plot._upload_url:
                    parts.append(f"![{plot.caption}]({plot._upload_url}?v={ts_bust})")
                else:
                    parts.append(f"- {plot.caption}: _{plot.path.name}_")

        return "\n".join(parts)

    async def _post_comment(self, body: str) -> int | None:
        """Create a new PR comment. Returns the comment ID or None."""
        try:
            client = await self._get_client()
            resp = await client.post(
                f"{self._base_url}/repos/{self._repo}/issues/{self._pr_number}/comments",
                json={"body": body},
            )
            if resp.status_code in (200, 201):
                data = resp.json()
                return data.get("id")
            _log.warning(f"POST comment failed: {resp.status_code} {resp.text}")
            return None
        except Exception as exc:
            _log.error(f"POST comment error: {exc}")
            return None

    async def _edit_comment(self, comment_id: int, body: str) -> bool:
        """Edit an existing PR comment. Returns True on success."""
        try:
            client = await self._get_client()
            resp = await client.patch(
                f"{self._base_url}/repos/{self._repo}/issues/comments/{comment_id}",
                json={"body": body},
            )
            if resp.status_code == 200:
                return True
            _log.warning(f"PATCH comment {comment_id} failed: {resp.status_code}")
            return False
        except Exception as exc:
            _log.error(f"PATCH comment error: {exc}")
            return False

    async def send_status(self, update: StatusUpdate) -> bool:
        """Post or edit the rolling PR comment with status table + alerts + plots."""
        body = self._build_status_body(update)

        # If we have a rolling comment, try to edit it
        if self._comment_id is not None:
            success = await self._edit_comment(self._comment_id, body)
            if success:
                return True
            # Edit failed (comment deleted?) -- fall back to new comment
            _log.info(
                f"Rolling comment {self._comment_id} edit failed, "
                f"creating new comment"
            )
            self._comment_id = None

        # Create new comment
        new_id = await self._post_comment(body)
        if new_id is not None:
            self._comment_id = new_id
            return True
        return False

    async def send_alert(self, alert: Alert) -> bool:
        """Post a new PR comment for an alert (never edits rolling comment)."""
        body = format_alert_message(alert)
        new_id = await self._post_comment(body)
        return new_id is not None
```

```python
    async def _upload_plot(
        self, plot: PlotAttachment, branch: str
    ) -> str | None:
        """Upload a plot file to the repo via GitHub Contents API.

        Returns the raw.githubusercontent.com URL on success, None on failure.
        The file is committed to the experiment branch so it persists.
        """
        import base64

        try:
            content = plot.path.read_bytes()
            encoded = base64.b64encode(content).decode()

            # Path in the repo: experiments/<name>/plots/<filename>
            upload_path = f"plots/{plot.path.name}"
            api_url = (
                f"{self._base_url}/repos/{self._repo}/contents/{upload_path}"
            )

            # Check if file exists (need SHA for update)
            client = await self._get_client()
            get_resp = await client.get(api_url, params={"ref": branch})
            sha = None
            if get_resp.status_code == 200:
                sha = get_resp.json().get("sha")

            payload: dict[str, Any] = {
                "message": f"watchdog: upload {plot.path.name}",
                "content": encoded,
                "branch": branch,
            }
            if sha:
                payload["sha"] = sha

            resp = await client.put(api_url, json=payload)
            if resp.status_code in (200, 201):
                data = resp.json()
                download_url = data.get("content", {}).get("download_url")
                if download_url:
                    _log.info(f"Plot uploaded: {download_url}")
                    return download_url
            _log.warning(f"Plot upload failed: {resp.status_code}")
            return None
        except Exception as exc:
            _log.error(f"Plot upload error: {exc}")
            return None

    @staticmethod
    def _cache_bust_url(url: str, timestamp: int) -> str:
        """Append cache-busting query parameter to a URL."""
        separator = "&" if "?" in url else "?"
        return f"{url}{separator}v={timestamp}"
```
