Fixture loading helper:
```python
import json
from pathlib import Path
import yaml

FIXTURE_DIR = Path(__file__).parent.parent / "fixtures" / "watchdog"

def load_fixture(name: str):
    """Load a watchdog test fixture by name (solo_hover, adversarial_heilbron, prompt_coevo)."""
    fixture_path = FIXTURE_DIR / name
    with open(fixture_path / "experiment.yaml") as f:
        manifest_data = yaml.safe_load(f)
    with open(fixture_path / "redis_data.json") as f:
        redis_data = json.load(f)
    metrics_data = None
    metrics_path = fixture_path / "metrics.yaml"
    if metrics_path.exists():
        with open(metrics_path) as f:
            metrics_data = yaml.safe_load(f)
    return manifest_data, redis_data, metrics_data
```

    runner = CliRunner()
    result = runner.invoke(cli, ["watchdog"])
    assert result.exit_code != 0

def test_watchdog_cmd_import_chain():

    assert result.exit_code == 0
